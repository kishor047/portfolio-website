# Simple data analysis
import numpy as np

def analyze(data):
    """Return mean and std deviation of a numeric list."""
    arr = np.array(data)
    return np.mean(arr), np.std(arr)

if __name__ == '__main__':
    sample = [12, 7, 22, 5, 18]
    mean, std = analyze(sample)
    print(f"Mean: {mean:.2f}  Std: {std:.2f}")
